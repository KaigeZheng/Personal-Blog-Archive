  Double-Width CAS Pointer

struct Pointer {
    void* ptr;
    uint64_t count;
};

template<typename T>
struct Node {
    T data;
    std::atomic<Node*> next;
    
    Node(T val) : data(val), next(nullptr) {}
    Node() : next(nullptr) {} // dummy node
};

template<typename T>
class LockFreeQueue {
    private:
        // ä½¿ç”¨__int128ä½œä¸ºåŒå®½CASæ•°æ®è½½ä½“
        typedef __int128 AtomicPointerType;

        // å¤´å°¾æŒ‡é’ˆå°è£…
        struct PtrCount {
            Node<T>* ptr;
            uint64_t count;
        };

        std::atomic<AtomicPointerType> head;
        std::atomic<AtomicPointerType> tail;

        // å°†PtrCountæ‰“åŒ…æˆ__int128
        static AtomicPointerType pack(PtrCount pc) {

            // ä½64ä½æ”¾æŒ‡é’ˆï¼Œ é«˜64ä½æ”¾è®¡æ•°å™¨
            uint64_t ptrVal = reinterpret_cast<uint64_t>(pc.ptr);

            val |= (AtomicPointerType)pc.count << 64;
            return val;
        }

        // å°†__int128æ‹†æˆPtrCount
        static PtrCount unpack(AtomicPointerType val) {
            PtrCount pc;
            pc.ptr = reinterpret_cast<Node<T>*>(static_cast<uint64_t>(val & 0xFFFFFFFFFFFFFFFF));
            pc.count = static_cast<uint64_t>(val >> 64);
            return pc;
        }
    public:
        LockFreeQueue() {
            Node<T>* dummy = new Node<T>();
            PtrCount pc{dummy, 0};
            head.store(pack(pc));
            tail.store(pack(pc));
        }

        ~LockFreeQueue() {
            PtrCount h = unpack(head.load());
            while(h.ptr != nullptr) {
                Node<T>* next = h.ptr->next.load();
                delete h.ptr;
                h.ptr = next;
            }
        }

        void enqueue(const T& value) {
            Node<T>* newNode = new Node<T>(value);
            PtrCount tailOld;

                // åŠ è½½å½“å‰çš„tailå¹¶è§£åŒ…ä¸ºtailOld
